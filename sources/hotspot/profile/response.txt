<script>
(function () {

  const clockEl = document.getElementById("server-time-text");
  if (!clockEl) return;

  // baseline waktu server (dari PHP)
  const BASE_SERVER_MS = new Date("2026-05-17T16:52:23+07:00").getTime();

  // timestamp browser saat halaman dimuat
  const BASE_BROWSER_MS = Date.now();

  let formatter = new Intl.DateTimeFormat("id-ID", {
    day: "2-digit",
    month: "2-digit",
    year: "numeric",
    hour: "2-digit",
    minute: "2-digit",
    second: "2-digit",
    hour12: false,
  });

  function renderClock() {

    // selisih waktu berjalan sejak halaman dimuat
    const elapsed = Date.now() - BASE_BROWSER_MS;

    // waktu server sekarang = baseline server + elapsed
    const nowServer = new Date(BASE_SERVER_MS + elapsed);

    // format manual agar hasil: Kam, 25-12-2025 19:24:16
    const parts = formatter.formatToParts(nowServer);

    let out = {};
    for (const p of parts) out[p.type] = p.value;

    clockEl.textContent =
      `${out.day}-${out.month}-${out.year} `
    + `${out.hour}:${out.minute}:${out.second}`;
  }

  // render awal + live tiap 1 detik
  renderClock();
  setInterval(renderClock, 1000);

})();
</script>
